  <script>
    let today = new Date();
    let currentMonth = today.getMonth();
    let currentYear = today.getFullYear();

    const events = {};

    function updateTime() {
      const now = new Date();
      document.getElementById("current-time").innerText = now.toLocaleTimeString();
    }

    function loadCalendar(month, year) {
      const calendar = document.getElementById("calendar");
      calendar.innerHTML = '';
      document.getElementById("calendar-title").innerText = `${new Date(year, month).toLocaleString('default', { month: 'long' })} ${year}`;

      const firstDay = new Date(year, month, 1).getDay();
      const daysInMonth = new Date(year, month + 1, 0).getDate();

      for (let i = 0; i < firstDay; i++) {
        const blank = document.createElement("div");
        calendar.appendChild(blank);
      }

      for (let date = 1; date <= daysInMonth; date++) {
        const cell = document.createElement("div");
        const cellDate = new Date(year, month, date);
        const dateStr = cellDate.toISOString().split('T')[0];

        cell.classList.add("calendar-day");
        if (cellDate.toDateString() === today.toDateString()) {
          cell.classList.add("today");
        } else if (cellDate < today) {
          cell.classList.add("past");
        } else {
          cell.classList.add("future");
        }

        cell.innerHTML = `<strong>${date}</strong><br><small>In: 08:30 | Out: 16:15</small><br><small>Status: Pending</small>`;

        if (!events[dateStr]) events[dateStr] = [];

        const tooltip = document.createElement("div");
        tooltip.classList.add("tooltip-data");
        tooltip.innerHTML = events[dateStr].map(e => `<div>${e.time} - ${e.title}</div>`).join('');
        cell.appendChild(tooltip);

        cell.addEventListener('click', () => {
          document.getElementById("eventDate").value = dateStr;
          document.getElementById("eventForm").reset();
          new bootstrap.Modal(document.getElementById('eventModal')).show();
        });

        calendar.appendChild(cell);
      }
    }

    function changeMonth(offset) {
      currentMonth += offset;
      if (currentMonth > 11) {
        currentMonth = 0;
        currentYear++;
      } else if (currentMonth < 0) {
        currentMonth = 11;
        currentYear--;
      }
      loadCalendar(currentMonth, currentYear);
    }

    $("#eventForm").on("submit", function(e) {
      e.preventDefault();
      const date = $("#eventDate").val();
      const title = $("#eventTitle").val();
      const time = $("#eventTime").val();
      events[date].push({ title, time });
      bootstrap.Modal.getInstance(document.getElementById('eventModal')).hide();
      loadCalendar(currentMonth, currentYear);
    });

    loadCalendar(currentMonth, currentYear);
    setInterval(updateTime, 1000);
  </script>

    // Cart functionality
    const cartButton = document.getElementById('cartButton');
    const cartCountSpan = document.getElementById('cartCount');
    const cartModal = document.getElementById('cartModal');
    const cartItemsUl = document.getElementById('cartItems');
    const closeCartBtn = cartModal.querySelector('.closeCart');
    const checkoutBtn = cartModal.querySelector('.checkoutBtn');

    let cart = JSON.parse(localStorage.getItem('cart')) || [];

    function updateCartCount() {
      const count = cart.length;
      cartCountSpan.textContent = count;
      cartCountSpan.style.display = count ? 'flex' : 'none';
    }
    updateCartCount();

    function renderCartItems() {
      cartItemsUl.innerHTML = '';
      let total = 0;
      if (!cart.length) {
        cartItemsUl.innerHTML = '<li>No books selected.</li>';
        checkoutBtn.disabled = true;
        checkoutBtn.style.opacity = 0.5;
        document.getElementById('cartTotal').textContent = 'Total: â‚¹0';
        return;
      }
      cart.forEach((book, i) => {
        total += parseFloat(book.price);
        const li = document.createElement('li');
        li.innerHTML = `<strong>${book.name}</strong><br>Price: â‚¹${book.price} | Avail: ${book.availability}<br>
          <button data-index="${i}" class="removeBtn" style="margin-top:5px;">Remove</button>`;
        cartItemsUl.appendChild(li);
      });
      document.getElementById('cartTotal').textContent = `Total: â‚¹${total.toFixed(2)}`;
      checkoutBtn.disabled = false;
      checkoutBtn.style.opacity = 1;
      cartItemsUl.querySelectorAll('.removeBtn').forEach(btn => {
        btn.addEventListener('click', e => {
          cart.splice(parseInt(e.target.dataset.index), 1);
          localStorage.setItem('cart', JSON.stringify(cart));
          updateCartCount();
          renderCartItems();
        });
      });
    }

    cartButton.addEventListener('click', () => {
      if (cartModal.style.display === 'flex') cartModal.style.display = 'none';
      else { renderCartItems(); cartModal.style.display = 'flex'; }
    });
    closeCartBtn.addEventListener('click', () => cartModal.style.display = 'none');
    checkoutBtn.addEventListener('click', () => {
      if (!cart.length) return;
      const cartData = encodeURIComponent(JSON.stringify(cart));
      alert('Redirecting to GovPay...');
      window.location.href = `https://govpay.example.com/checkout?cart=${cartData}`;
    });

    // Receive message from iframe to add to cart
    window.addEventListener('message', event => {
      const data = event.data;
      if (data?.type === 'addToCart') {
        if (!cart.find(b => b.name === data.book.name)) {
          cart.push(data.book);
          localStorage.setItem('cart', JSON.stringify(cart));
          updateCartCount();
          alert(`Added "${data.book.name}" to cart!`);
        } else {
          alert(`"${data.book.name}" is already in the cart.`);
        }
      }
    });

        `${new Date(year, month).toLocaleString('default', { month: 'long' })} ${year}`;

        calendar.appendChild(blank);

      for (let date = 1; date <= daysInMonth; date++) {
        const cellDate = new Date(year, month, date);
        const dateStr = cellDate.toISOString().split('T')[0];
        const cell = document.createElement("div");
        cell.classList.add("calendar-day");

        if (cellDate.toDateString() === today.toDateString()) {
          cell.classList.add("today");
        } else if (cellDate < today) {
          cell.classList.add("past");
        } else {
          cell.classList.add("future");
        }

        const savedEvent = events[dateStr] ? `<br><small>Event: ${events[dateStr]}</small>` : "";
        cell.innerHTML = `<strong>${date}</strong>${savedEvent}`;

        if (cellDate >= today) {

              loadCalendar(currentMonth, currentYear);
            }
          });
        }

        calendar.appendChild(cell);
      }
    }

<script>
  const categories = ['thesis','news','procurement','other'];

  function getData(cat){ return JSON.parse(localStorage.getItem(cat)) || []; }
  function saveData(cat, data){ localStorage.setItem(cat, JSON.stringify(data)); }

  function renderForm(cat){
    document.getElementById('form-container').innerHTML = `
      <form id="form-${cat}">
        <input type="text" id="title" placeholder="Title" required />
        <textarea id="desc" placeholder="Description" rows="3" required></textarea>
        <label><strong>Upload Photos (up to 10):</strong></label><br/>
        <input type="file" id="photos" accept="image/*" multiple /><div class="thumbs" id="thumbs"></div>
        <label><strong>Upload PDFs:</strong></label><br/>
        <input type="file" id="pdfs" accept="application/pdf" multiple /><br/><br/>
        <button type="submit">Submit</button>
      </form>
    `;

    const form = document.getElementById(`form-${cat}`);
    const thumbs = document.getElementById('thumbs');

    form.photos.addEventListener('change', e => {
      thumbs.innerHTML = '';
      [...e.target.files].slice(0, 10).forEach(file => {
        const reader = new FileReader();
        reader.onload = ev => {
          const img = document.createElement('img');
          img.src = ev.target.result;
          thumbs.appendChild(img);
        };
        reader.readAsDataURL(file);
      });
    });

    form.onsubmit = e => {
      e.preventDefault();
      const title = form.title.value.trim();
      const desc = form.desc.value.trim();
      const photos = [...form.photos.files].slice(0,10);
      const pdfs = [...form.pdfs.files];
      const readerPromises = photos.map(f => new Promise(res => {
        const r = new FileReader();
        r.onload = ev => res({ name: f.name, data: ev.target.result });
        r.readAsDataURL(f);
      }));
      const pdfPromises = pdfs.map(f => new Promise(res => {
        const r = new FileReader();
        r.onload = ev => res({ name: f.name, data: ev.target.result });
        r.readAsDataURL(f);
      }));

      Promise.all([...readerPromises, ...pdfPromises]).then(items => {
        const data = getData(cat);
        data.push({
          id: Date.now(),
          title, desc,
          photos: items.slice(0, photos.length),
          pdfs: items.slice(photos.length),
          status: 'pending'
        });
        saveData(cat, data);
        form.reset(); 
        thumbs.innerHTML = '';
        renderList();
      });
    };
  }

  function renderList(){
    const list = document.getElementById('itemList');
    list.innerHTML = '';
    categories.forEach(cat => {
      getData(cat).forEach(item => {
        const li = document.createElement('li');

        // Build image thumbnails HTML
        const imagesHtml = item.photos && item.photos.length > 0
          ? `<div class="preview-images">${item.photos.map(p => `<img src="${p.data}" alt="${p.name}">`).join('')}</div>`
          : '';

        // Build PDF links HTML
        const pdfsHtml = item.pdfs && item.pdfs.length > 0
          ? `<div class="preview-pdfs">` +
            item.pdfs.map(p => `<a href="${p.data}" download="${p.name}" target="_blank">${p.name}</a>`).join(' | ') +
            `</div>`
          : '';

        li.innerHTML = `
          <strong>[${cat.toUpperCase()}] ${item.title}</strong>
          <span class="status ${item.status === 'approved' ? 'approved' : ''}">${item.status}</span>
          <p>${item.desc}</p>
          ${imagesHtml}
          ${pdfsHtml}
        `;

        const delBtn = document.createElement('button');
        delBtn.className = 'delete';
        delBtn.textContent = 'Delete';
        delBtn.onclick = () => {
          const arr = getData(cat).filter(i => i.id !== item.id);
          saveData(cat, arr);
          renderList();
        };

        const approveBtn = document.createElement('button');
        approveBtn.className = 'approve';
        approveBtn.textContent = item.status === 'pending' ? 'Approve' : 'Unapprove';
        approveBtn.onclick = () => {
          item.status = item.status === 'pending' ? 'approved' : 'pending';
          const arr = getData(cat).map(i => i.id === item.id ? item : i);
          saveData(cat, arr);
          renderList();
        };

        li.appendChild(delBtn);
        li.appendChild(approveBtn);
        list.appendChild(li);
      });
    });
  }

  document.querySelectorAll('input[name="cat"]').forEach(r => {
    r.addEventListener('change', () => renderForm(r.value));
  });

  renderForm('thesis');
  renderList();
</script>

    async function loadProfiles(role) {
      const container = document.getElementById('profilesContainer');
      container.innerHTML = 'Loading...';

      try {
        const response = await fetch('nara_profiles.json');
        const data = await response.json();

        const filtered = data.filter(profile => profile.role === role);

        if (filtered.length === 0) {
          container.innerHTML = `<p>No staff found for role: ${role}</p>`;
          return;
        }

        container.innerHTML = '';
        filtered.forEach(profile => {
          const card = document.createElement('div');
          card.className = 'profile-card';

          card.innerHTML = `
            <img src="${profile.photo}" alt="${profile.name}">
            <h3>${profile.name}</h3>
            <p><strong>${profile.position}</strong></p>
            <p>${profile.department}</p>
            <p>${profile.location}</p>
            <a href="mailto:${profile.email}">${profile.email}</a>
          `;

          container.appendChild(card);
        });
      } catch (error) {
        container.innerHTML = `<p>Error loading profiles: ${error.message}</p>`;
      }
    }

    // Initial load
    const roleSelect = document.getElementById('roleSelect');
    loadProfiles(roleSelect.value);

    // Handle role change
    roleSelect.addEventListener('change', () => {
      loadProfiles(roleSelect.value);
    });
  </script>
